81      private TripleIterator findTriples(String spoQuery) throws TrippiException {
82  
83          // get results from adaptor, wrapped in our own TripleIterator
84          Connection conn = null;
85          try {
86              conn = _pool.getConnection();
87              conn.setAutoCommit(false);
88              QueryResults results = _adaptor.query(conn, 
89                                                    QueryLanguage.SPO, 
90                                                    _fetchSize,
91                                                    true, // autoRelease
92                                                    spoQuery);
93              return new MPTTripleIterator(results);
94          } catch (Exception e) { 
95              if (conn != null) {
96                  try { conn.close(); } catch (Exception e2) { }
97              }
98              throw new TrippiException("Error querying triples", e);
99          }
100     }

118     private void update(Set<Triple> triples, boolean delete) 
119             throws TrippiException {
120         Connection conn = null;
121         boolean startedTransaction = false;
122         boolean success = false;
123         try {
124             conn = _pool.getConnection();
125             conn.setAutoCommit(false);
126             startedTransaction = true;
127 
128             Set<org.nsdl.mptstore.rdf.Triple> mptSet = jrdfToMPT(triples);
129 
130             if (delete) {
131                 _adaptor.deleteTriples(conn, mptSet.iterator());
132             } else {
133                 _adaptor.addTriples(conn, mptSet.iterator());
134             }
135             success = true;
136 
137         } catch (Exception e) {
138             throw new TrippiException("Error updating triples", e);
139         } finally {
140             if (conn != null) {
141                 if (startedTransaction) {
142                     if (!success) {
143                         try {
144                             conn.rollback();
145                         } catch (Exception e2) { }
146                     }
147                     try {
148                         conn.commit();
149                         conn.setAutoCommit(true);
150                     } catch (Exception e2) { }
151                 }
152                 try { conn.close(); } catch (Exception e) { }
153             }
154         }

158      * Convert the given set of JRDF Triple objects to 
159      * MPT Triple objects.

161     protected static Set<org.nsdl.mptstore.rdf.Triple> jrdfToMPT(Set<Triple> jrdfTriples) {
162         Set<org.nsdl.mptstore.rdf.Triple> mptSet = new HashSet<org.nsdl.mptstore.rdf.Triple>(jrdfTriples.size());
163         Iterator<Triple> iter = jrdfTriples.iterator();
164         while (iter.hasNext()) {
165             Triple jrdfTriple = iter.next();
166             org.nsdl.mptstore.rdf.SubjectNode mptSubject =
167                     (org.nsdl.mptstore.rdf.SubjectNode) 
168                     jrdfToMPT(jrdfTriple.getSubject());
169             org.nsdl.mptstore.rdf.PredicateNode mptPredicate =
170                     (org.nsdl.mptstore.rdf.PredicateNode) 
171                     jrdfToMPT(jrdfTriple.getPredicate());
172             org.nsdl.mptstore.rdf.ObjectNode mptObject =
173                     (org.nsdl.mptstore.rdf.ObjectNode) 
174                     jrdfToMPT(jrdfTriple.getObject());
175             mptSet.add(new org.nsdl.mptstore.rdf.Triple(
176                     mptSubject, mptPredicate, mptObject));
177         }
178         return mptSet;
179     }

181     protected static org.nsdl.mptstore.rdf.Node jrdfToMPT(Node jrdfNode) {
182         try {
183             if (jrdfNode instanceof URIReference) {
184                 URIReference jrdfURIReference = (URIReference) jrdfNode;
185                 return new org.nsdl.mptstore.rdf.URIReference(jrdfURIReference.getURI());
186             } else if (jrdfNode instanceof Literal) {
187                 Literal jrdfLiteral = (Literal) jrdfNode;
188                 String lang = jrdfLiteral.getLanguage();
189                 if (lang != null && lang.length() > 0) {
190                     return new org.nsdl.mptstore.rdf.Literal(
191                             jrdfLiteral.getLexicalForm(), lang);
192                 } else if (jrdfLiteral.getDatatypeURI() != null) {
193                     org.nsdl.mptstore.rdf.URIReference dType =
194                             new org.nsdl.mptstore.rdf.URIReference(
195                             jrdfLiteral.getDatatypeURI());
196                     return new org.nsdl.mptstore.rdf.Literal(
197                             jrdfLiteral.getLexicalForm(), dType);
198                 } else {
199                     return new org.nsdl.mptstore.rdf.Literal(
200                             jrdfLiteral.getLexicalForm());
201                 }
202             } else {
203                 throw new RuntimeException("Unrecognized node type; cannot "
204                         + "convert to MPT Node: " + jrdfNode.getClass().getName());
205             }
206         } catch (URISyntaxException e) {
207             throw new RuntimeException("Bad URI syntax, cannot convert to "
208                     + "MPT Node", e);
209         } catch (ParseException e) {
210             throw new RuntimeException("Bad language syntax, cannot convert to "
211                     + "MPT Node", e);
212         }
213     }
