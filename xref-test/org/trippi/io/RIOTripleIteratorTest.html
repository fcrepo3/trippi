94      @Test
95      public void testFromStream() throws Exception {
96          InputStream in = new ByteArrayInputStream(rdf.getBytes());
97          TripleIterator iter = m_factory.fromStream(in, RDFFormat.RDF_XML);
98          HashMap<String, Triple> triples = new HashMap<String, Triple>();
99          while(iter.hasNext()) {
100             Triple next = iter.next();
101             triples.put(next.getPredicate().toString(), next);
102         }
103         String subject =
104                 "info:fedora/test:pid";
105         String contentModel =
106                 "info:fedora/fedora-system:def/model#hasContentModel";
107         String memberOf =
108                 "info:fedora/fedora-system:def/relations-external#isMemberOf";
109         String testProp = "info:fedora/fedora-system:def/model#testProperty";
110         Triple actual = triples.remove(contentModel);
111         assertTriple(actual, subject, contentModel, "info:fedora/demo:CmodelForBMech_DualResImageImpl");
112         actual = triples.remove(memberOf);
113         assertTriple(actual, subject, memberOf, "info:fedora/demo:SmileyStuff");
114         actual = triples.remove(testProp);
115         // this property is a literal
116         assertTriple(actual, subject, testProp, "\"test\"");
117         assertEquals("Unexpected remainder in test triples", 1, triples.size());
118     }

158     @Test
159     public void testInterruptingStream() throws Exception {
160         InputStream in = new ByteArrayInputStream(rdf.getBytes());
161         RIOTripleIterator iter =
162                 (RIOTripleIterator) m_factory.fromStream(in, RDFFormat.RDF_XML);
163         iter.next();
164         iter.next();
165         iter.close();
166         // because we look ahead one triple to support hasNext, the tripleCount
167         // is one higher than the number of triples iterated over
168         // but should still be less than the total
169         assertEquals("Could not cut short iteration", 3, iter.m_tripleCount);
170     }

172     private void assertTriple(Triple actual, String subject, String predicate,
