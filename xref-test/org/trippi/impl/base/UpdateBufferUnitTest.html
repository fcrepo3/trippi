95      // Test that adding to buffer can occur at the same time,
96      // on a different thread from the flush
97  
98      public void testCanAddDuringFlush() throws Exception {
99  
100         // set up our fake session
101         FakeTriplestoreSession session = new FakeTriplestoreSession(1000);
102 
103         // set up our buffer with an add
104         _buffer = getBuffer(10, 5);
105         _buffer.add(getTriple(1, 1, 1));
106 
107         // prepare a separate thread to perform the flushing
108         FlushingThread flusher = new FlushingThread(_buffer, session, 1);
109         flusher.stopAfterNextFlush();
110 
111         // pause the session so it locks the flusher till we're ready
112         session.setPaused(true);
113 
114         // start the flusher thread
115         flusher.start();
116 
117         // wait for the session to pause the thread
118         while (session.getPausingThread() == null) {
119             try { Thread.sleep(25); } catch (InterruptedException e) { }
120         }
121 
122         // if the buffer impl blocks this while session.add is being
123         // invoked, this will catch it.  the FakeTriplestoreSession
124         // will timeout, the flusher will fail and set a flag for us,
125         // the buffer's block will be released, and this call will
126         // continue.
127         _buffer.add(getTriple(1, 1, 2));
128 
129         // un-pause the session...this should allow the flusher to complete
130         session.setPaused(false);
131 
132         // wait a full second if we have to
133         long waitedMillis = 0;
134         while (flusher.isAlive() && waitedMillis < 1000) {
135             try {
136                 Thread.sleep(100);
137                 waitedMillis += 100;
138             } catch (InterruptedException e) {
139             }
140         }
141 
142         assertFalse("Flusher thread didn't complete even after "
143                 + "un-pausing the fake session", flusher.isAlive());
144 
145         // check whether the flusher detected an error.
146         // if it did, it was due to a lock imposed by the buffer impl,
147         // and this test should fail.
148         Exception flushError = flusher.getError();
149         if (flushError != null) {
150             fail("Flusher reported error during flush: " 
151                     + flushError.getClass().getName() + ": " 
152                     + flushError.getMessage());
153         }
154 
155     }

228         FakeTriplestoreSession session = new FakeTriplestoreSession();

259     /**
260      * On the given session, do the indicated mods with multiple modder
261      * threads and asynchronous flushing.
262      *
263      * This method returns when all threads are done.
264      */
265     private void doConcurrentMods(FakeTriplestoreSession session,
266                                   int numModders,
267                                   boolean useBatchCalls,
268                                   int chunksPerModder,
269                                   int triplesPerChunk,
270                                   boolean adds,
271                                   int flushThreshold) throws Exception {
272 
273         _buffer = getBuffer(flushThreshold, flushThreshold);
274 
275         // init + start the flusher
276         FlushingThread flusher = new FlushingThread(_buffer, session, flushThreshold);
277         flusher.start();
278 
279         // init + start the modders
280         ModdingThread[] modders = new ModdingThread[numModders];
281         for (int i = 0; i < numModders; i++) {
282             modders[i] = new ModdingThread(i,
283                                            _buffer,
284                                            useBatchCalls,
285                                            chunksPerModder,
286                                            triplesPerChunk,
287                                            adds);
288             modders[i].start();
289         }
290 
291         // wait for modders to finish
292         int numModdersFinished = 0;
293         while (numModdersFinished < numModders) {
294             numModdersFinished = 0;
295             for (int i = 0; i < numModders; i++) {
296                 if (!modders[i].isAlive()) {
297                     numModdersFinished++;
298                 }
299             }
300         }
301 
302         // wait for flusher to deal with overflow
303         while (_buffer.size() >= flushThreshold) {
304             try { Thread.sleep(10); } catch (InterruptedException e) { }
305         }
306 
307         // signal to flusher to do one more if needed, then finish
308         flusher.stopAfterNextFlush();
309 
310         // wait for flusher to finish
311         while (flusher.isAlive()) {
312             try { Thread.sleep(10); } catch (InterruptedException e) { }
313         }
314 
315         // check modders for any errors, and throw first found
316         for (int i = 0; i < numModders; i++) {
317             Exception e = modders[i].getError();
318             if (e != null) {
319                 throw new Exception("Unexpected error in modder thread", e);
320             }
321         }
322

497     public class FlushingThread extends Thread {
498 
499         private UpdateBuffer _buffer;
500         private TriplestoreSession _session;
501         private Exception _error;
502         private boolean _stopAfterNextFlush;
503         private int _flushIfThisSizeOrAbove;
504 
505         public FlushingThread(UpdateBuffer buffer,
506                               TriplestoreSession session,
507                               int flushIfThisSizeOrAbove) {
508             _buffer = buffer;
509             _session = session;
510             _flushIfThisSizeOrAbove = flushIfThisSizeOrAbove;
511         }
512 
513         public void stopAfterNextFlush() {
514             _stopAfterNextFlush = true;
515         }
516 
517         // flush at least once, and then keep flushing until 
518         // stopAfterNextFlush is called
519         @Override
520 		public void run() {
521             try {
522                 boolean finish = false;
523                 while (!finish) {
524                     if (_stopAfterNextFlush) {
525                         finish = true;
526                         while (_buffer.size() > 0) {
527                             _buffer.flush(_session);
528                             Thread.yield();
529                         }
530                     } else {
531                         if (_buffer.size() >= _flushIfThisSizeOrAbove) {
532                             _buffer.flush(_session);
533                         }
534                         Thread.yield();
535                     }
536                 }
537             } catch (Exception e) {
538                 _error = e;
539             }
540         }
541 
542         public Exception getError() {
543             return _error;
544         }

547     public class ModdingThread extends Thread {
548 
549         private int _id;
550         private UpdateBuffer _buffer;
551         private boolean _useBatchCalls;
552         private int _numChunks;
553         private int _triplesPerChunk;
554         private boolean _doAdds;
555 
556         private Exception _error;
557 
558         public ModdingThread(int id, 
559                              UpdateBuffer buffer,
560                              boolean useBatchCalls, 
561                              int numChunks,
562                              int triplesPerChunk, 
563                              boolean doAdds) {
564             _id = id;
565             _buffer = buffer;
566             _useBatchCalls = useBatchCalls;
567             _numChunks = numChunks;
568             _triplesPerChunk = triplesPerChunk;
569             _doAdds = doAdds;
570 
571         }
572 
573         @Override
574 		public void run() {
575 
576             try {
577                 for (int i = 0; i < _numChunks; i++) {
578          
579                     // generate triples for this chunk
580                     List<Triple> triples = new ArrayList<Triple>();
581                     for (int j = 0; j < _triplesPerChunk; j++) {
582                         triples.add(getTriple(_id, i, j));
583                     }
584 
585                     // do the adds or deletes for this chunk
586                     if (_doAdds) {
587                         if (_useBatchCalls) {
588                             _buffer.add(triples);
589                         } else {
590                             Iterator<Triple> iter = triples.iterator();
591                             while (iter.hasNext()) {
592                                 Triple triple = iter.next();
593                                 _buffer.add(triple);
594                             }
595                         }
596                     } else {
597                         if (_useBatchCalls) {
598                             _buffer.delete(triples);
599                         } else {
600                             Iterator<Triple> iter = triples.iterator();
601                             while (iter.hasNext()) {
602                                 Triple triple = iter.next();
603                                 _buffer.delete(triple);
604                             }
605                         }
606                     }
607 
608                     // yield so we're not hogging any impl-specific
609                     // buffer locks
610                     Thread.yield();
611                 }
612             } catch (Exception e) {
613                 _error = e;
614             }
615         }
616 
617         private Exception getError() {
618             return _error;
619         }
