35      /**
36       * Get an iterator over the triples in the given stream.
37       *
38       * The baseURI is used to resolve any relative URI references.
39       * If given as null, http://localhost/ will be used.
40       */
41      public TripleIterator fromStream(InputStream in,
42                                              String baseURI,
43                                              RDFFormat format) throws TrippiException {
44          if (baseURI == null) baseURI = "http://localhost/";
45          org.openrdf.rio.RDFParser parser;
46          if (format == RDFFormat.RDF_XML) {
47              parser = new org.openrdf.rio.rdfxml.RDFXMLParser();
48          } else if (format == RDFFormat.TURTLE) {
49              parser = new org.openrdf.rio.turtle.TurtleParser();
50          } else if (format == RDFFormat.N_TRIPLES) {
51              parser = new org.openrdf.rio.ntriples.NTriplesParser();
52          } else {
53              throw new TrippiException("Unsupported input format: " + format.getName());
54          }
55          return new RIOTripleIterator(in, parser, baseURI, m_executor);
56      }
57  
58      /**
59       * Get an iterator over the triples in the given stream.
60       */
61      public TripleIterator fromStream(InputStream in,
62                                              RDFFormat format) throws TrippiException {
63          return fromStream(in, null, format);
64      }
